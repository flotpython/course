import asyncio
#Task.all_tasks() et Task.current_task() remplacées par asyncio.all_tasks() et asyncio.current_task()

async def foo():
    pass

async def main():
    # On crée une task avec asyncio.create_task() On peut lui donner un nom (option)
    task = asyncio.create_task(foo(), name="Foo")
    print("La task Foo est en attente")
    print(task)
    print("---------")
    print(f"Task Foo terminée ? : {task.done()}") #La task est terminée ?
    print("---------")
    print(f"Task en cours :\n{asyncio.current_task()}") # retourne la task en cours.
    print("---------")
    tasks = asyncio.all_tasks() # retourne un set de tasks non terminées.
    print(f"Set des tasks non terminées :\n{tasks}")
    await task
    print("---------")
    print(f"Task Foo terminée ? : {task.done()}") #La task est terminée ?
    
asyncio.run(main())
